39 def generate(about):
40     with open("index.py") as codefile:
41         code = codefile.read()
42 
43     with open("index_tpl.html") as tpl:
44         template = Template(tpl.read())
45 
46     formatter = formatters.HtmlFormatter(linenos="inline", style="gruvbox")
47     about["style"] = formatter.get_style_defs()
48     about["body"] = highlight(code, lexers.PythonLexer(), formatter)
49 
50     index_html = template.substitute(about)
51 
52     with open("index.html", "w") as index_html_file:
53         index_html_file.write(index_html)
