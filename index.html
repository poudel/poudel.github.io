52 def generate(about):
53     formatter = HtmlFormatter(linenos='inline', style='gruvbox')
54 
55     with open('index.py') as codefile:
56         code = codefile.read()
57 
58     with open('index_tpl.html') as tpl:
59         template = Template(tpl.read())
60 
61     about['style'] = formatter.get_style_defs()
62     about['body'] = highlight(code, PythonLexer(), formatter)
63 
64     index_html = template.substitute(about)
65 
66     with open('index.html', 'w') as index_html_file:
67         index_html_file.write(index_html)
